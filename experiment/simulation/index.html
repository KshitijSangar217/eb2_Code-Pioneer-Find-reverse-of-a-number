#include <stdio.h>
int main()
{
int num, remainder = 0, reverse_num = 0;
while(num>0)
{
remainder = num % 10;
reverse_num = 10 * reverse_num;
reverse_num = reverse_num + remainder;
num = num / 10;
}
printf("Reversed number : %d", reverse_num);
return 0;
}
